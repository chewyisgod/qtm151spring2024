dataset = pd.read_csv("data_quiz/wdi_ageprop_2020.csv")
list_varnames = ["percent_ages0to14","percent_ages15to64"] 

for names in list_varnames:
    column = dataset[names].to_list()
    plt.hist(x = column)
    plt.xlabel("Age")
    plt.ylabel("Frequency")
    plt.title("Graph of " + str(names))
    plt.show()
